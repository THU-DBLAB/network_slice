# install git
sudo apt-get install -y git

# clone mininet
cd ~
git clone git://github.com/mininet/mininet

# install mininet and ovs 2.10.5
sudo ./mininet/util/install.sh -n3V 2.10.5

# check the new ovs version
sudo ovs-vsctl -V

# test mininet
sudo mn

exit
